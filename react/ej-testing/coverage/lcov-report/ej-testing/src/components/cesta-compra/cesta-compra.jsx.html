import React from 'react';
import Producto from './producto';
 
class CestaCompra extends React.Component {
  constructor() {
    super();
    this.state = {
      productos: [
        { id: 1, nombre: 'One Plus 5T' },
        { id: 2, nombre: 'Funda One Plus' },
        { id: 3, nombre: 'SD 64GB' }
      ]
    },
    this.getListaProductos = this.getListaProductos.bind(this);
    this.getCestaCompra = this.getCestaCompra.bind(this);
    this.getLogin = this.getLogin.bind(this);
  }
 
  getListaProductos() {
    return this.state.productos.map(producto => (<Producto key={producto.id} id={producto.id} nombre={producto.nombre} />));
  }
 
  getCestaCompra() {
    let productos = this.getListaProductos();
    return (
      <div>
        <h1>Cesta compra</h1>
        { productos.length > 0 ? productos : <p>Cesta vacia</p> }
        <button type="button">Logout</button>
      </div>
    );
  }
 
  getLogin() {
    return (
      <div>
        <h1>No estas logueado</h1>
        <button type="button">Login</button>
      </div>
    );
  }
 
  render() {
    const cmp = this.props.logueado ? this.getCestaCompra() : this.getLogin();
    return (
     cmp
    );
  }
}
 
export default CestaCompra;
